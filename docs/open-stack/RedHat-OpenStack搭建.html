    # 第一章 VMware Workstation 虚拟机环境

    # 1.1 NTP

    RHEL7.1:NTP-Server+YUM 源(http):192.168.11.253/24

    # 1.2 server-a

    RHEL7.1:控制节点+计算节点:192.168.11.10/24

    # 1.3 server-b

    RHEL7.1:计算节点:192.168.11.11/24

    # 2.2.3 挂载系统安装光盘，建立临时本地 YUM 仓库，安装 Apache

    1. 建立临时目录

      mkdir /dvd
      1

    2. 挂载光盘

      mount /dev/sr0 /dvd
      1

    3. 建立本地 YUM 仓库

      vim /etc/yum.repos.d/dvd.repo
      1
      [DVD]
name=dvd
baseurl=file:///dvd
enabled=1
gpgcheck=0
      1
      2
      3
      4
      5

    4. 测试本地仓库可用性

      yum makecache ; yum repolist all
      1

    5. 安装 Apache

      yum -y install httpd
      1

    6. 启动 httpd 服务

      systemctl start httpd.service
      1

    7. 查看 httpd 服务状态

      systemctl status httpd.service
      1

    8. 设置 httpd 服务开机自启动

      systemctl enable httpd.service
      1

    # 2.3 搭建基于 http 的网络 YUM 源仓库

/iso/rhel-server-7.1-x86_64-dvd.iso /var/www/html/dvd iso9660 defaults,loop,ro 0 0

    # 2.4 传递仓库指向文件

    利用 SSH 的客户端程序 SCP 将仓库指向文件传送至控制节点(192.168.11.10)和计算节点 (192.168.11.11)

    [root@ntp ~]# scp /etc/yum.repos.d/dvd.repo root@192.168.11.10:/etc/yum.repos.d/dvd.repo 
[root@ntp ~]# scp /etc/yum.repos.d/dvd.repo root@192.168.11.11:/etc/yum.repos.d/dvd.repo 
[root@ntp ~]# scp /etc/yum.repos.d/openstack.repo root@192.168.11.10:/etc/yum.repos.d/openstack.repo 
[root@ntp ~]# scp /etc/yum.repos.d/openstack.repo root@192.168.11.11:/etc/yum.repos.d/openstack.repo

    # 第五章 安装红帽 OpenStack

    # 利用布署工具生成自动应答文件(文件名可任意)

    packstack --gen-answer-file=/root/ans.txt
    1

    # 修改自动应答文件实现自动化布署

    [root@server-a ~]# vim /root/ans.txt #注意以下几项
CONFIG_NTP_SERVERS=192.168.11.253       #NTP 服务器 IP
CONFIG_CONTROLLER_HOST=192.168.11.10     #控制节点 IP
CONFIG_COMPUTE_HOSTS=192.168.11.10,192.168.11.11 (建立多计算节点，此处控制节点复用)       #计算节点 IP
CONFIG_NETWORK_HOSTS=192.168.11.10 (暂时由控制节点担当此角色)   #网络节点 IP
CONFIG_KEYSTONE_ADMIN_PW=redhat      #WEB 界面管理员口令
CONFIG_HORIZON_SSL=y                  #加密通信 WEB 界面
(默认“n”，即 http://，修改“y”，即 https://) 
CONFIG_PROVISION_DEMO=n              #不创建演示环境，默认为"y"
创建实例(演示时可采用)，修改"n"，不创建实例(利于学习管理过程)
